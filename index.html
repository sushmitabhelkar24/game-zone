// 🚀 JavaScript Functions Practice (All-in-One File) // 1. Add Two Numbers function add(a, b) { return a + b; } console.log("Add:", add(3, 5)); // 2. Even or Odd function checkEvenOdd(num) { return num % 2 === 0 ? "Even" : "Odd"; } console.log("Even/Odd:", checkEvenOdd(7)); // 3. Maximum of Two Numbers function max(a, b) { return a > b ? a : b; } console.log("Max:", max(10, 20)); // 4. Square of a Number function square(n) { return n * n; } console.log("Square:", square(4)); // 5. Greeting Function function greet(name) { console.log("Hello, " + name); } greet("Rahul"); // 6. Sum of Array function sumArray(arr) { let sum = 0; for (let i = 0; i < arr.length; i++) { sum += arr[i]; } return sum; } console.log("Sum of array:", sumArray([1, 2, 3, 4])); // 7. Factorial function factorial(n) { let result = 1; for (let i = 1; i <= n; i++) { result *= i; } return result; } console.log("Factorial:", factorial(5)); // 8. Positive / Negative / Zero function checkNumber(num) { if (num > 0) { return "Positive"; } else if (num < 0) { return "Negative"; } else { return "Zero"; } } console.log("Check number:", checkNumber(-3)); // 🔥 Bonus Arrow Functions const addArrow = (a, b) => a + b; const squareArrow = n => n * n; const isEven = n => n % 2 === 0; console.log("Arrow Add:", addArrow(2, 3)); console.log("Arrow Square:", squareArrow(5)); console.log("Arrow isEven:", isEven(6));
